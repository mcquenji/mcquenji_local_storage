Implementation

@nonVirtual
Future<File> resolveFile(String file) async {
  final dir = await resolveAppDataDir();
  await dir.create(recursive: true);

  final path = '${dir.path}/$file';

  log('Resolved file path: $path');

  final f = File(path);
  await f.create();

  return f;
}
