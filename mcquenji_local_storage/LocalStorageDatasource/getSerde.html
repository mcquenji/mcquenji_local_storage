@nonVirtual
@protected
IGenericSerializer<T, JSON> getSerde<T>() {
  try {
    return Modular.get<LocalStorageSerializer<T>>();
  } on Exception catch (e) {
    throw LocalStorageException('No serializer found for $T: $e. Did you forget registering it?');
  }
}
